from collections import Counter
import statistics

# Sample array of numbers
numbers = [2, 4, 4, 4, 5, 5, 6, 6, 7, 8, 8, 9]

# Mean calculation
mean = sum(numbers) / len(numbers)

# Median calculation
median = statistics.median(numbers)

# Mode calculation
mode = statistics.mode(numbers)

# Output
print("Mean:", mean)
print("Median:", median)
print("Mode:", mode)
